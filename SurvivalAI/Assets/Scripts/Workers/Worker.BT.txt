tree("Root")
	fallback
		tree("Flee")
		tree("Rest")
		tree("GatherResources")

tree("Flee")
	sequence
		EnemyNearby
		RunToHero
		IsArrivedAtDestination

tree("Rest")
	sequence
		IsTired
		WalkToMainBuilding
		IsArrivedAtDestination
		RestAtMainBuilding
		IsRested

tree("GatherResources")
	while not HasEnoughResources
		sequence
			ChooseResource
			ChooseTool
			WalkToClosestResource
			IsArrivedAtDestination
			not EnemyNearby
			GatherResource
			WaitForGatheringEnd
			WalkToClosestWarehouse
			IsArrivedAtDestination
			DepositResources

tree("Root")
	fallback
		tree("Flee")
		tree("Rest")
		tree("Build")
		tree("GatherResources")


tree("Flee")
	sequence
		IsThreatened
		RunToHero


tree("Rest")
	sequence
		IsTired
		fallback
			tree("WalkToMainBuilding")
			tree("RestAtMainBuilding")
		TryToMate

tree("WalkToMainBuilding")
	sequence
		WalkToMainBuilding
		IsWalking
	
tree("RestAtMainBuilding")
	sequence
		not IsWalking
		RestAtMainBuilding


tree("Build")
	sequence
		CanBuild
		WalkToReadyBuilding
		Build

tree("GatherResources")
	sequence
		not HasEnoughResources
		fallback
			tree("ChooseResource")
			tree("WalkToResource")
			tree("CollectResource")
			tree("WalkToWarehouse")
			tree("DepositResource")


tree("ChooseResource")
	sequence
		not IsGathering
		ChooseResource
		ChooseTool

tree("WalkToResource")
	sequence
		IsGathering
		not IsFull
		WalkToClosestResource
		IsWalking

tree("CollectResource")
	sequence
		IsGathering
		not IsWalking
		not IsFull
		GatherResource

tree("WalkToWarehouse")
	sequence
		IsGathering
		IsFull
		WalkToClosestWarehouse
		IsWalking

tree("DepositResource")
	sequence
		IsFull
		not IsWalking
		DepositResources
